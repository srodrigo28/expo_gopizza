* Criando o projeto 
```
expo init gopizza
```

* Entrar na pasta do projeto
```
cd go pizza
```

* Criar o arquivo tsconfig.json em branco
```
expo start
```

* Para rodar no android
```
npm run android
```

* Auxiliar de diret√≥rio
```
babel-plugin-module-resolver -D
```

* babel.config.json
```
module.exports = function(api) {
  api.cache(true);
  return {
    presets: ['babel-preset-expo'],
    plugins:[
      [
        'module-resolver',
        {
          root: ['./src'],
          extensions: [
            '.ts', '.tsx', '.js',
            '.json','.css'
          ],
          alias: {
            '@components': './src/components',
            '@screens': './src/screens',
            '@assets': './src/assets',
          }
        }
      ]
    ]
  }
}
```

* Configurar tsconfig.json
```
{
  "extends": "expo/tsconfig.base",
  "compilerOptions": {
    "strict": true,
    "baseUrl": "/",
    "paths": {
      "@src/*": [ "./src/*"],
      "@assets/*": ["./src/assets/*"],
      "@components/*": ["./src/components/*"],
      "@screens/*": ["./src/screens/*"],
    }
  },
}
```
